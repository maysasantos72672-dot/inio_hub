local BotModule = {}

function BotModule.IsAlive(Character)
    if not Character then return false end
    local Humanoid = Character:FindFirstChild("Humanoid")
    if not Humanoid then return false end
    return Humanoid.Health > 0
end

function BotModule.FireRemote(Name, ...)
    if CommF then
        return CommF:InvokeServer(Name, ...)
    end
end

function BotModule.GetChar()
    return lp.Character
end

function BotModule.GetMobs(range)
    local char = BotModule.GetChar()
    if not char or not char:FindFirstChild("HumanoidRootPart") then
        return {}
    end
    
    local root = char.HumanoidRootPart
    local mobs = {}
    
    if not workspace:FindFirstChild("Enemies") then return mobs end
    
    for _, mob in ipairs(workspace.Enemies:GetChildren()) do
        if mob and mob.Parent then
            local hrp = mob:FindFirstChild("HumanoidRootPart")
            local hum = mob:FindFirstChild("Humanoid")
            
            if hrp and hum and hum.Health > 0 then
                if (hrp.Position - root.Position).Magnitude <= (range or 60) then
                    table.insert(mobs, mob)
                end
            end
        end
    end
    
    return mobs
end

function BotModule.BringMob(mob)
    -- Proteção: para se o mob sumir ou morrer antes/durante o processo
    if not mob or not mob.Parent then return end
    local hum = mob:FindFirstChild("Humanoid")
    if not hum or hum.Health <= 0 then return end
    
    local mobHRP = mob:FindFirstChild("HumanoidRootPart")
    if not mobHRP then return end
    
    local char = BotModule.GetChar()
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    
    local playerRoot = char.HumanoidRootPart
    local bringSpeed = Settings.BringMobSpeed / 10
    
    pcall(function()
        if not mobHRP or not playerRoot then return end
        local direction = (playerRoot.Position - mobHRP.Position).Unit
        local newPos = mobHRP.Position + direction * bringSpeed * 5
        
        mobHRP.CFrame = CFrame.new(newPos)
        
        if mobHRP:FindFirstChild("BodyVelocity") then
            mobHRP.BodyVelocity.Velocity = direction * bringSpeed * 100
        else
            local BV = Instance.new("BodyVelocity")
            BV.Velocity = direction * bringSpeed * 100
            BV.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
            BV.Parent = mobHRP
            game:GetService("Debris"):AddItem(BV, 0.1)
        end
    end)
end

function BotModule.SuperFastAttackULTRA(enemyHitBox, bladeHits)
    if not enemyHitBox or not enemyHitBox.Parent then return end
    local enemy = enemyHitBox.Parent
    if not Settings.SuperClique then return end
    
    -- Correção: Variável local definida e escopada corretamente antes de rodar os loops
    local times = 10
    
    table.insert(bladeHits, {enemy, enemyHitBox})
    table.insert(bladeHits, math.random())
    
    local startTime = tick()
    while (tick() - startTime) < 1 and Settings.SuperClique do
        -- Proteção extra caso o inimigo morra dentro desse ciclo de 1 segundo
        if not enemyHitBox or not enemyHitBox.Parent then break end
        
        for i = 1, times do
            if not Settings.SuperClique then return end
            
            if attackRemote then
                pcall(function() attackRemote:FireServer() end)
            end
            
            if hitRemote then
                pcall(function() hitRemote:FireServer(enemyHitBox, bladeHits, nil, nil) end)
            end
        end
        task.wait() -- Impede o congelamento da thread/crash do script
    end
end

function BotModule.SuperFastAttackINSANE(enemyHitBox, bladeHits)
    if not enemyHitBox or not enemyHitBox.Parent then return end
    local enemy = enemyHitBox.Parent
    if not Settings.SuperClique then return end
    
    table.insert(bladeHits, {enemy, enemyHitBox})
    table.insert(bladeHits, math.random())
    
    if attackRemote then
        for i = 1, 100 do
            if not Settings.SuperClique then return end
            if not enemyHitBox or not enemyHitBox.Parent then break end
            pcall(function() attackRemote:FireServer() end)
        end
    end
    
    if hitRemote then
        for i = 1, 100 do
            if not Settings.SuperClique then return end
            if not enemyHitBox or not enemyHitBox.Parent then break end
            pcall(function() hitRemote:FireServer(enemyHitBox, bladeHits, nil, nil) end)
        end
    end
end

task.spawn(function()
    while task.wait(0.001) do
        if Settings.Enabled and Settings.SuperClique then
            local mobs = BotModule.GetMobs(80)
            
            for _, mob in ipairs(mobs) do
                if not Settings.SuperClique then break end
                
                -- Proteção: Ignora o mob se ele sumir repentinamente no loop ipairs
                if not mob or not mob.Parent then continue end
                local hum = mob:FindFirstChild("Humanoid")
                if not hum or hum.Health <= 0 then continue end
                
                local hrp = mob:FindFirstChild("HumanoidRootPart")
                if hrp then
                    if Settings.FastAttackMode == "INSANE" then
                        BotModule.SuperFastAttackINSANE(hrp, {})
                    else
                        BotModule.SuperFastAttackULTRA(hrp, {})
                    end
                end
            end
        end
    end
end)

task.spawn(function()
    while task.wait(0.001) do
        if Settings.Enabled and Settings.BringMob then
            local mobs = BotModule.GetMobs(Settings.BringMobRange)
            
            for _, mob in ipairs(mobs) do
                if not Settings.BringMob then break end
                
                -- Proteção: Evita processar inimigos destruídos ou inválidos
                if not mob or not mob.Parent then continue end
                local hum = mob:FindFirstChild("Humanoid")
                if not hum or hum.Health <= 0 then continue end
                
                BotModule.BringMob(mob)
            end
        end
    end
end)

task.spawn(function()
    while task.wait(0.1) do
        if Settings.Enabled and Settings.AutoFarm then
            local mobs = BotModule.GetMobs(Settings.AutoFarmRange)
            
            for _, mob in ipairs(mobs) do
                if not mob or not mob.Parent then continue end
                local humanoid = mob:FindFirstChild("Humanoid")
                
                if humanoid and humanoid.Health > 0 then
                    local hrp = mob:FindFirstChild("HumanoidRootPart")
                    if hrp then
                        if Settings.FastAttackMode == "INSANE" then
                            BotModule.SuperFastAttackINSANE(hrp, {})
                        else
                            BotModule.SuperFastAttackULTRA(hrp, {})
                        end
                    end
                end
            end
        end
    end
end)

task.spawn(function()
    while task.wait(1) do
        if Settings.Enabled and Settings.AutoBuso then
            local Character = Player.Character
            local IsAlive = BotModule.IsAlive(Character)
            
            if IsAlive and not Character:FindFirstChild("HasBuso") then
                -- Correção: Método nativo correto usando o CollectionService do Roblox
                if CollectionService:HasTag(Character, "Buso") then
                    BotModule.FireRemote("Buso")
                elseif Money.Value >= 25000 then
                    BotModule.FireRemote("BuyHaki", "Buso")
                end
            end
        end
    end
end)
